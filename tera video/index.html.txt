  <script>
    const input = document.getElementById("videoLink");
    const iframe = document.getElementById("videoFrame");
    const likedList = document.getElementById("likedList");
    const likedPanel = document.getElementById("likedPanel");

    function applyTheme(mode) {
      if (mode === "light") document.body.className = "light";
      else if (mode === "dark") document.body.className = "";
      else {
        const prefersDark = window.matchMedia("(prefers-color-scheme: dark)").matches;
        document.body.className = prefersDark ? "" : "light";
      }
      localStorage.setItem("theme", mode);
    }

    function switchTheme() {
      const mode = document.getElementById("themeToggle").value;
      applyTheme(mode);
    }

    function toggleDataSaver() {
      const isOn = document.getElementById("dataSaverToggle").checked;
      localStorage.setItem("dataSaver", isOn ? "true" : "false");
    }

    function resumeLast() {
      const saved = localStorage.getItem("lastLink");
      if (saved) {
        input.value = saved;
        prepareVideo();
      }
    }

    function prepareVideo() {
      const link = input.value.trim();
      if (!link.includes("terabox.com")) {
        alert("‚ö†Ô∏è Please paste a valid Terabox link.");
        return;
      }
      localStorage.setItem("lastLink", link);
      const dataSaver = localStorage.getItem("dataSaver") === "true";
      if (dataSaver) {
        document.getElementById("manualLoadBtn").style.display = "inline-block";
      } else {
        iframe.src = link;
        iframe.style.display = "block";
        iframe.style.opacity = "1";
      }
      updateShareButtons(link);
    }

    function loadIframe() {
      const link = input.value.trim();
      iframe.src = link;
      iframe.style.display = "block";
      iframe.style.opacity = "1";
      updateShareButtons(link);
    }

    function enableMini() {
      iframe.classList.add("mini");
      document.getElementById("backBtn").style.display = "inline-block";
    }

    function disableMini() {
      iframe.classList.remove("mini");
      document.getElementById("backBtn").style.display = "none";
    }

    function likeVideo() {
      const link = input.value.trim();
      if (!link.includes("terabox.com")) {
        alert("Paste a valid Terabox link first.");
        return;
      }
      let liked = JSON.parse(localStorage.getItem("likedLinks") || "[]");
      if (!liked.includes(link)) {
        liked.push(link);
        localStorage.setItem("likedLinks", JSON.stringify(liked));
        alert("‚ù§Ô∏è Saved to Liked Videos!");
      } else {
        alert("Already liked üëç");
      }
    }

    function showLiked() {
      likedPanel.style.display = "block";
      likedList.innerHTML = "";
      const liked = JSON.parse(localStorage.getItem("likedLinks") || "[]");
      if (liked.length === 0) {
        likedList.innerHTML = "<li>No liked videos yet.</li>";
        return;
      }
      liked.forEach(link => {
        const item = document.createElement("li");
        item.innerHTML = `<a href="#" style="color:#00c3ff;" onclick="playFromList('${link}')">${link}</a>`;
        likedList.appendChild(item);
      });
    }

    function playFromList(link) {
      input.value = link;
      prepareVideo();
    }

    function updateShareButtons(link) {
      document.getElementById("tgShare").href = `https://t.me/share/url?url=${encodeURIComponent(link)}`;
      document.getElementById("waShare").href = `https://api.whatsapp.com/send?text=${encodeURIComponent(link)}`;
      document.getElementById("shareSection").style.display = "block";
    }

    function copyLink() {
      const link = input.value.trim();
      navigator.clipboard.writeText(link);
      alert("üìã Copied to clipboard!");
    }

    window.onload = () => {
      const savedTheme = localStorage.getItem("theme") || "system";
      document.getElementById("themeToggle").value = savedTheme;
      applyTheme(savedTheme);

      const savedLink = localStorage.getItem("lastLink");
      if (savedLink) {
        input.value = savedLink;
        document.getElementById("resumeBtn").style.display = "inline-block";
      }

      const dataSaver = localStorage.getItem("dataSaver") === "true";
      document.getElementById("dataSaverToggle").checked = dataSaver;

      // Clipboard auto-paste if terabox link exists
      navigator.clipboard.readText().then(text => {
        if (text.includes("terabox.com")) {
          input.value = text;
          console.log("üìã Auto-pasted link from clipboard.");
        }
      }).catch(err => {});
    }
